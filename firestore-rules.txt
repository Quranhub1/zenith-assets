rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Allow anyone to read courses (top-level collection)
    match /RESOURCES_STUDYPEDIA/{courseId} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.token.email == 'kaigwaakram123@gmail.com';
      
      // semesters subcollection
      match /semesters/{semesterId} {
        allow read: if request.auth != null;
        allow write: if request.auth != null && request.auth.token.email == 'kaigwaakram123@gmail.com';
        
        // courseunits nested subcollection
        match /courseunits/{unitId} {
          allow read: if request.auth != null;
          allow write: if request.auth != null && request.auth.token.email == 'kaigwaakram123@gmail.com';
          
          // documents nested under courseunits
          match /documents/{docId} {
            allow read: if request.auth != null;
            allow write: if request.auth != null && request.auth.token.email == 'kaigwaakram123@gmail.com';
          }
        }
        
        // documents subcollection under semesters
        match /documents/{docId} {
          allow read: if request.auth != null;
          allow write: if request.auth != null && request.auth.token.email == 'kaigwaakram123@gmail.com';
        }
      }
    }
    
    // ZENITH RESOURCES - For Zenith Assets investment platform
    // Path: ZENITH RESOURCES/{docId}/users, deposits, withdrawals, investments, transactions
    match /ZENITH%20RESOURCES/{resourceId} {
      // Allow read access to everyone (for viewing data)
      allow read: if true;
      // Allow write access only to admin
      allow write: if request.auth != null && request.auth.token.email == 'kaigwaakram123@gmail.com';
      
      // Subcollections under ZENITH RESOURCES/{resourceId}
      match /users/{userId} {
        allow read: if true;
        allow write: if request.auth != null && request.auth.token.email == 'kaigwaakram123@gmail.com';
      }
      
      match /deposits/{depositId} {
        allow read: if true;
        allow write: if request.auth != null && request.auth.token.email == 'kaigwaakram123@gmail.com';
      }
      
      match /withdrawals/{withdrawalId} {
        allow read: if true;
        allow write: if request.auth != null && request.auth.token.email == 'kaigwaakram123@gmail.com';
      }
      
      match /investments/{investmentId} {
        allow read: if true;
        allow write: if request.auth != null && request.auth.token.email == 'kaigwaakram123@gmail.com';
      }
      
      match /transactions/{transactionId} {
        allow read: if true;
        allow write: if request.auth != null && request.auth.token.email == 'kaigwaakram123@gmail.com';
      }
    }
    
    // Allow collection-group queries for latest documents
    match /{path=**}/documents/{docId} {
      allow read: if request.auth != null;
    }
    
    // User payments - users can read their own, admin can read all
    match /user_payments/{userId} {
      allow read: if request.auth != null && request.auth.uid == userId 
                  || request.auth.token.email == 'kaigwaakram123@gmail.com';
      allow write: if request.auth != null && request.auth.token.email == 'kaigwaakram123@gmail.com';
    }
    
    // Allow authenticated users to read their own data
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Pesapal orders
    match /pesapal_orders/{orderId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.token.email == 'kaigwaakram123@gmail.com';
    }
  }
}
