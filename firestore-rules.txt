rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // ZENITH RESOURCES - OPEN FOR ALL (anyone can read/write)
    match /ZENITH%20RESOURCES/{resourceId} {
      allow read: if true;
      allow write: if true;
      
      match /users/{userId} {
        allow read: if true;
        allow write: if true;
      }
      
      match /deposits/{depositId} {
        allow read: if true;
        allow write: if true;
      }
      
      match /withdrawals/{withdrawalId} {
        allow read: if true;
        allow write: if true;
      }
      
      match /investments/{investmentId} {
        allow read: if true;
        allow write: if true;
      }
      
      match /transactions/{transactionId} {
        allow read: if true;
        allow write: if true;
      }
    }
    
    // User payments - open for all
    match /user_payments/{userId} {
      allow read: if true;
      allow write: if true;
    }
    
    // Allow authenticated users to read their own data - open for all
    match /users/{userId} {
      allow read: if true;
      allow write: if true;
    }
}
